Data


new Vue({
	el: '#example-data',
	template: '
{{message}}
',
	data: function() {
		return {
			message: 'Hello, VueJS!'
		}
	}
});

Methods


new Vue({
	el: '#example-methods',
	template: '
{{message}}
',
	data: function() {
		return {
			message: 'Hello, VueJS!'
		}
	},
	methods: {
		handleClick: function(e){
			this.message += ' Now with interactivity!';
		}
	}
});
