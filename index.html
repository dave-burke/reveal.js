Vue.js

An Incrementally Adoptable JavaScript Component Library

Basic usage


<script src="https://unpkg.com/vue@2.5.13/dist/vue.js"></script>



<div id="example"></div>

Multi-line Template Strings


new Vue({
  el: '#example',
  template: '
Hello, VueJS!
'
});

Multi-line Template Strings


new Vue({
  el: '#example',
  template: '
' +
    'Hello, VueJS!' +
    '
'
});

Multi-line Template Strings


new Vue({
  el: '#example',
  template: '
\
    Hello, VueJS!\
    
'
});

Multi-line Template Strings


new Vue({
  el: '#example',
  template: `
    

      Hello, VueJS!
    

  `
});

Data


new Vue({
  el: '#example',
  template: '
{{message}}
',
  data: {
    message: 'Hello, VueJS!'
  }
});

Attribute binding


.green {
  color: green;
}
            

new Vue({
  el: '#example',
  template: '
Hello, VueJS!
',
  data: {
    className: 'green'
  }
});

Methods


new Vue({
  el: '#example',
  template: `

      Hello, VueJS!
    
`,
  data: {
    className: 'normal'
  },
  methods: {
    handleClick: function(e){
      this.className = 'green';
    }
  }
});

Shortcuts


<div v-on:click="handleClick" v-bind:class="className">
Is exactly the same as 

<div @click="handleClick" :class="className">

Two-way form binding


new Vue({
  el: '#example',
  template: `

    
    
{{text}}

  
`,
  data: {
    text: 'text'
  }
});

Conditional rendering


new Vue({
  el: '#example',
  template: `

    
    Yes
    <span v-else>No</span>
  
`,
  data: {
    isChecked: false
  }
});

Looping


new Vue({
  el: '#example',
  template: `
    
    
  • {{item}}
    • 
  
`,
  data: {
    items: ['one','two','three']
  }
});

Components


Vue.component('welcome', {
  template: '
{{message}}
',
  data: function() {
    return {
      message: 'Hello from a Vue component!'
    }
  }
});
new Vue({
  el: '#example',
  template: ''
});

Props


Vue.component('welcome', {
  props: ['message'],
  template: '
{{message}}
',
});
new Vue({
  el: '#example',
  template: ``
});

Single-file Components

Welcome.vue 

<template>
  <div v-bind:class="className">Hello, VueJS!</div>
</template>
<script>
module.exports = {
  data: () => ({
    className: 'green'
  })
}
</script>
<style scoped>
.green {
  color: green;
}
</style>
